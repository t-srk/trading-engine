cmake_minimum_required(VERSION 3.20)
project(TradingEngine VERSION 0.1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(MSVC)
    add_compile_options(/W4 /WX- /permissive- /wd4100 /wd4702)
else()
    add_compile_options(-Wall -Wextra -Wpedantic -Wshadow)
endif()

# ── Dependencies ──────────────────────────────────────────────────────────────
find_package(Boost REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)

# ── Shared engine sources ─────────────────────────────────────────────────────
set(ENGINE_SOURCES
    src/order_book.cpp
    src/matching_engine.cpp
)

# ── Main executable ───────────────────────────────────────────────────────────
add_executable(trading_engine
    src/main.cpp
    src/session.cpp
    src/server.cpp
    ${ENGINE_SOURCES}
)

target_include_directories(trading_engine PRIVATE include)
target_link_libraries(trading_engine PRIVATE
    Boost::boost
    nlohmann_json::nlohmann_json
)

# ── Tests ─────────────────────────────────────────────────────────────────────
add_executable(tests
    tests/test_order_book.cpp
    ${ENGINE_SOURCES}
)

target_include_directories(tests PRIVATE include)
